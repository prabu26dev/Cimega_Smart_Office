rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // ── HELPER FUNCTIONS ──────────────────────────────────
    
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function getUserData() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data;
    }

    function isAdmin() {
      return isAuthenticated() && (getUserData().role == 'admin' || getUserData().roles.hasAny(['admin']));
    }

    function isRole(roleName) {
      return isAuthenticated() && (getUserData().role == roleName || getUserData().roles.hasAny([roleName]));
    }

    function isOpsOrAdmin() {
      return isAdmin() || isRole('ops');
    }

    // ── COLLECTIONS ───────────────────────────────────────

    // USERS: Admin can manage all; Users can read all (to see names/schools) but only edit self (limited).
    match /users/{userId} {
      allow read: if isAuthenticated();
      allow write: if isAdmin() || request.auth.uid == userId;
    }

    // SEKOLAH: Admin can manage; All authenticated can read.
    match /sekolah/{sekolahId} {
      allow read: if isAuthenticated();
      allow write: if isAdmin();
    }

    // KATEGORI & KONTEN (Administrasi)
    // Admin can manage everything.
    // Users can read if the category/content is visible to their role.
    match /kategori/{katId} {
      allow read: if isAuthenticated() && (isAdmin() || getUserData().roles.hasAny(resource.data.visibleTo));
      allow write: if isAdmin();
    }

    match /konten/{kontenId} {
      allow read: if isAuthenticated() && (isAdmin() || getUserData().roles.hasAny(resource.data.visibleTo));
      allow write: if isAdmin();
    }

    // ADMIN TEMPLATES (AI)
    match /admin_templates/{tmpId} {
      allow read: if isAuthenticated();
      allow write: if isAdmin();
    }

    // MUSIC (BGM)
    match /app_music/{musicId} {
      allow read: if true; // Publicly readable for BGM
      allow write: if isAdmin();
    }

    // CONFIG & UPDATES
    match /appConfig/{docId} {
      allow read: if true;
      allow write: if isAdmin();
    }
    
    match /updateHistory/{docId} {
      allow read: if isAuthenticated();
      allow write: if isAdmin();
    }

    // NOTIFICATIONS
    match /notifications/{notifId} {
      allow read: if isAuthenticated() && (resource.data.target == 'all' || isRole(resource.data.target) || isAdmin());
      allow write: if isAdmin();
    }

    // LOGS
    match /adminLog/{logId} {
      allow create: if isAuthenticated(); // Allow logging from app
      allow read, update, delete: if isAdmin();
    }

    // ── MODULE BASED RULES (Legacy/Extensions) ───────────
    
    match /students/{studentId} {
      allow read: if isAuthenticated();
      allow write: if isOpsOrAdmin();
    }

    match /ptk/{ptkId} {
      allow read: if isAuthenticated();
      allow write: if isOpsOrAdmin();
    }

    match /keuangan/{docId} {
      allow read, write: if isAdmin() || isRole('bendahara') || isRole('kepsek');
    }

    // GLOBAL CATCH-ALL (DISABLED IN PRODUCTION)
    // Currently set to allow read if authenticated for stability, but write is strictly for Admin.
    match /{document=**} {
      allow read: if isAuthenticated();
      allow write: if isAdmin();
    }
  }
}
